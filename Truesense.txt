#include <Wire.h>
#include <VL53L0X.h> // MUST BE THE POLOLU LIBRARY
#include <Adafruit_PWMServoDriver.h>

// ==========================================
// CONFIGURATION DASHBOARD
// ==========================================
const bool SWAP_AXES = false;      
const bool REVERSE_TREMOR = false; 

// --- PCA9685 Servo Driver ---
Adafruit_PWMServoDriver pwm = Adafruit_PWMServoDriver();
#define SERVO_MIN 150 
#define SERVO_MAX 600 
#define NEUTRAL_POS 375 

// --- VL53L0X Laser Array ---
#define NUM_LASERS 5 // Change to 4 if D2 is still disconnected
const int XSHUT_PINS[NUM_LASERS] = {2, 3, 4, 5, 6}; 
const int LASER_ADDR[NUM_LASERS] = {0x30, 0x31, 0x32, 0x33, 0x34}; 
VL53L0X lasers[NUM_LASERS];
int currentLaser = 0; // For Round-Robin reading

// --- MPU6050 Variables ---
const int MPU_ADDR = 0x68; 
float gyroX_offset = 0, gyroY_offset = 0;
float accPitch_offset = 0, accRoll_offset = 0;
float roll = 0, pitch = 0;

// --- Timing & PD Control ---
unsigned long prevTime;
float dt;

// AGGRESSIVE STABILIZATION TUNING 
float Kp = 1.0;  // Muscle: 1:1 reaction to the tremor
float Kd = 0.25; // Shock Absorber: Smooths out the aggressive motor snaps
float targetAngle = 0; 
float errorPitch, prevErrorPitch, outputPitch;

void setup() {
  Serial.begin(115200);
  delay(1000); 
  Serial.println("\n--- NEUROGRIP FULL SYSTEM BOOTING ---");
  
  Wire.begin();

  // ==========================================
  // 1. SILENT SERVO BOOT
  // ==========================================
  Serial.println("1. Starting PCA9685 (Silent Boot)...");
  pwm.begin();
  pwm.setPWMFreq(50); 
  delay(100);

  // ==========================================
  // 2. BOOT SEQUENCE: VL53L0X LASERS
  // ==========================================
  Serial.println("2. Booting Laser Array...");
  // Reset all lasers by pulling XSHUT LOW
  for (int i = 0; i < NUM_LASERS; i++) {
    pinMode(XSHUT_PINS[i], OUTPUT);
    digitalWrite(XSHUT_PINS[i], LOW); 
  }
  delay(100);

  // Turn them on ONE BY ONE and change their I2C address
  for (int i = 0; i < NUM_LASERS; i++) {
    Serial.print("   -> Turning on Laser on Pin "); Serial.println(XSHUT_PINS[i]);
    digitalWrite(XSHUT_PINS[i], HIGH); 
    delay(50); 
    lasers[i].init();
    lasers[i].setTimeout(500);
    lasers[i].setAddress(LASER_ADDR[i]); 
  }

  // ==========================================
  // 3. MPU6050 SILICON RESET & WAKE-UP
  // ==========================================
  Serial.println("3. Resetting MPU6050 Silicon...");
  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x6B); 
  Wire.write(0x80);    
  Wire.endTransmission(true);
  delay(150);

  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x6B); 
  Wire.write(0x00);    
  Wire.endTransmission(true);
  delay(100);

  bool mpuReady = false;
  while (!mpuReady) {
    Wire.beginTransmission(MPU_ADDR);
    Wire.write(0x3B);
    Wire.endTransmission(true); 
    Wire.requestFrom(MPU_ADDR, 14, true);

    int16_t test_ax = Wire.read() << 8 | Wire.read();
    int16_t test_ay = Wire.read() << 8 | Wire.read();
    int16_t test_az = Wire.read() << 8 | Wire.read();
    
    if (test_ax != 0 && test_ax != -1) mpuReady = true;
    delay(10);
  }
  Serial.println("   -> MPU is Awake!");

  // ==========================================
  // 4. MPU6050 AUTO-CALIBRATION
  // ==========================================
  Serial.println("4. Calibrating MPU... KEEP GLOVE FLAT & STILL!");
  delay(500);

  for (int i = 0; i < 500; i++) {
    Wire.beginTransmission(MPU_ADDR);
    Wire.write(0x3B);
    Wire.endTransmission(true); 
    Wire.requestFrom(MPU_ADDR, 14, true);

    int16_t ax = Wire.read() << 8 | Wire.read();
    int16_t ay = Wire.read() << 8 | Wire.read();
    int16_t az = Wire.read() << 8 | Wire.read();
    Wire.read(); Wire.read(); 
    int16_t gx = Wire.read() << 8 | Wire.read();
    int16_t gy = Wire.read() << 8 | Wire.read();
    int16_t gz = Wire.read() << 8 | Wire.read();

    if (SWAP_AXES) { int16_t temp = ax; ax = ay; ay = temp; temp = gx; gx = gy; gy = temp; }

    // FLOAT CAST: Prevents Math Crash
    float f_ax = ax; float f_ay = ay; float f_az = az;
    accRoll_offset += (atan2(f_ay, f_az) * 180 / PI);
    accPitch_offset += (atan2(-f_ax, sqrt(f_ay * f_ay + f_az * f_az)) * 180 / PI);
    gyroX_offset += (gx / 131.0);
    gyroY_offset += (gy / 131.0);
    delay(3);
  }

  accRoll_offset /= 500.0; accPitch_offset /= 500.0;
  gyroX_offset /= 500.0; gyroY_offset /= 500.0;

  Serial.println("--- SYSTEM READY --- Starting NeuroGrip Engine.");
  
  // WAKE UP SERVOS STAGGERED BEFORE LOOP
  for (int i = 0; i < 5; i++) {
    pwm.setPWM(i, 0, NEUTRAL_POS); 
    delay(200); 
  }
  
  prevTime = millis();
}

void loop() {
  unsigned long currentTime = millis();
  dt = (currentTime - prevTime) / 1000.0; 
  prevTime = currentTime;
  if (dt <= 0) return; 

  // ==========================================
  // A. READ TREMOR (Fast Cycle)
  // ==========================================
  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x3B); 
  Wire.endTransmission(true); 
  Wire.requestFrom(MPU_ADDR, 14, true); 

  int16_t ax = Wire.read() << 8 | Wire.read();
  int16_t ay = Wire.read() << 8 | Wire.read();
  int16_t az = Wire.read() << 8 | Wire.read();
  Wire.read(); Wire.read(); 
  int16_t gx = Wire.read() << 8 | Wire.read();
  int16_t gy = Wire.read() << 8 | Wire.read();
  int16_t gz = Wire.read() << 8 | Wire.read();

  if (ax == 0 && ay == 0 && az == 0) return; 

  if (SWAP_AXES) { int16_t temp = ax; ax = ay; ay = temp; temp = gx; gx = gy; gy = temp; }

  float f_ax = ax; float f_ay = ay; float f_az = az;
  float accAnglePitch = (atan2(-f_ax, sqrt(f_ay * f_ay + f_az * f_az)) * 180 / PI) - accPitch_offset;
  float gyroRatePitch = (gy / 131.0) - gyroY_offset;

  pitch = 0.98 * (pitch + gyroRatePitch * dt) + 0.02 * accAnglePitch;

  // ==========================================
  // B. STABILIZATION MATH (PD Loop)
  // ==========================================
  errorPitch = targetAngle - pitch;
  float dErrPitch = (errorPitch - prevErrorPitch) / dt;
  outputPitch = (Kp * errorPitch) + (Kd * dErrPitch);
  prevErrorPitch = errorPitch;

  float compensation = REVERSE_TREMOR ? -outputPitch : outputPitch;

  if (abs(compensation) < 1.5) {
    compensation = 0; 
  }

  int pulseLength = map(90 + compensation, 0, 180, SERVO_MIN, SERVO_MAX);
  pulseLength = constrain(pulseLength, SERVO_MIN, SERVO_MAX);

  for (int i = 0; i < 5; i++) {
    pwm.setPWM(i, 0, pulseLength); 
  }

  // ==========================================
  // C. READ FINGER THERAPY DATA (Round-Robin)
  // ==========================================
  int dist = lasers[currentLaser].readRangeSingleMillimeters();
  
  // Format output for the Serial Plotter
  Serial.print("Target:0,");
  Serial.print("Tremor:"); Serial.print(pitch); Serial.print(",");
  Serial.print("Servo_Out:"); Serial.print(compensation); Serial.print(",");
  
  // Print the active laser reading
  Serial.print("Finger_"); Serial.print(currentLaser + 1); Serial.print("_mm:"); 
  Serial.println(dist);

  // Cycle to the next laser for the next loop to keep the math blazing fast
  currentLaser++;
  if (currentLaser >= NUM_LASERS) currentLaser = 0;
}
