#include <Wire.h>
#include <Servo.h>

// --- Servo Setup ---
Servo servoPitch; // Up/Down (Pin 9)
Servo servoRoll;  // Left/Right (Pin 10)

const int pitchPin = 9;
const int rollPin = 10;

// --- MPU6050 Variables ---
const int MPU_ADDR = 0x68; 
int16_t accX, accY, accZ;
int16_t gyroX, gyroY, gyroZ;

// Calibration Offsets
float gyroX_offset = 0, gyroY_offset = 0;
float accPitch_offset = 0, accRoll_offset = 0;

float roll = 0, pitch = 0;

// --- Timing Variables ---
unsigned long prevTime;
float dt;

// --- PD Control Variables ---
// ⚠️ TUNE THESE TWO NUMBERS TONIGHT ⚠️
float Kp = 1.0;  // Proportional (Muscle): Try 0.8 to 1.5
float Kd = 0.08; // Derivative (Shock Absorber): Try 0.05 to 0.15

float targetAngle = 0; // We want the spoon perfectly flat (0 degrees)

float errorPitch, prevErrorPitch, outputPitch;
float errorRoll, prevErrorRoll, outputRoll;

void setup() {
  Serial.begin(115200); 
  
  // 1. Attach Servos and Center Them
  servoPitch.attach(pitchPin);
  servoRoll.attach(rollPin);
  servoPitch.write(90);
  servoRoll.write(90);

  // 2. Wake up MPU6050
  Wire.begin();
  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x6B); // Power management register
  Wire.write(0);    // Wake up the MPU-6050
  Wire.endTransmission(true);
  
  delay(1000); // Wait for physical servos to stop shaking

  // ==========================================
  // 3. THE HACKATHON AUTO-CALIBRATION ROUTINE
  // ==========================================
  Serial.println("CALIBRATING... DO NOT TOUCH THE SPOON!");
  
  for (int i = 0; i < 500; i++) {
    Wire.beginTransmission(MPU_ADDR);
    Wire.write(0x3B);
    Wire.endTransmission(false);
    Wire.requestFrom(MPU_ADDR, 14, true);

    accX = Wire.read() << 8 | Wire.read();
    accY = Wire.read() << 8 | Wire.read();
    accZ = Wire.read() << 8 | Wire.read();
    Wire.read(); Wire.read(); // Skip temp
    gyroX = Wire.read() << 8 | Wire.read();
    gyroY = Wire.read() << 8 | Wire.read();
    gyroZ = Wire.read() << 8 | Wire.read();

    float tempAccRoll = (atan2(accY, accZ) * 180 / PI);
    float tempAccPitch = (atan2(-accX, sqrt(accY * accY + accZ * accZ)) * 180 / PI);

    accRoll_offset += tempAccRoll;
    accPitch_offset += tempAccPitch;
    gyroX_offset += (gyroX / 131.0);
    gyroY_offset += (gyroY / 131.0);
    
    delay(3);
  }

  // Calculate the average offset
  accRoll_offset /= 500.0;
  accPitch_offset /= 500.0;
  gyroX_offset /= 500.0;
  gyroY_offset /= 500.0;

  Serial.println("Calibration Complete!");
  
  prevTime = millis();
}

void loop() {
  // 1. Calculate Time Elapsed (dt)
  unsigned long currentTime = millis();
  dt = (currentTime - prevTime) / 1000.0; 
  prevTime = currentTime;
  if (dt <= 0) return; 

  // 2. Read Raw Data from MPU6050
  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x3B); 
  Wire.endTransmission(false);
  Wire.requestFrom(MPU_ADDR, 14, true); 

  accX = Wire.read() << 8 | Wire.read();
  accY = Wire.read() << 8 | Wire.read();
  accZ = Wire.read() << 8 | Wire.read();
  Wire.read(); Wire.read(); 
  gyroX = Wire.read() << 8 | Wire.read();
  gyroY = Wire.read() << 8 | Wire.read();
  gyroZ = Wire.read() << 8 | Wire.read();

  // 3. Math: Accelerometer Angles (Minus Offsets)
  float rawAccRoll = (atan2(accY, accZ) * 180 / PI);
  float rawAccPitch = (atan2(-accX, sqrt(accY * accY + accZ * accZ)) * 180 / PI);
  
  float accAngleRoll = rawAccRoll - accRoll_offset;
  float accAnglePitch = rawAccPitch - accPitch_offset;

  // 4. Math: Gyro Rate (Minus Offsets)
  float gyroRateRoll = (gyroX / 131.0) - gyroX_offset;
  float gyroRatePitch = (gyroY / 131.0) - gyroY_offset;

  // 5. Complementary Filter (The Anti-Drift Engine)
  roll = 0.98 * (roll + gyroRateRoll * dt) + 0.02 * accAngleRoll;
  pitch = 0.98 * (pitch + gyroRatePitch * dt) + 0.02 * accAnglePitch;

  // 6. PD Control Loop: PITCH (Up/Down)
  errorPitch = targetAngle - pitch;
  float dErrPitch = (errorPitch - prevErrorPitch) / dt;
  outputPitch = (Kp * errorPitch) + (Kd * dErrPitch);
  prevErrorPitch = errorPitch;

  // 7. PD Control Loop: ROLL (Left/Right)
  errorRoll = targetAngle - roll;
  float dErrRoll = (errorRoll - prevErrorRoll) / dt;
  outputRoll = (Kp * errorRoll) + (Kd * dErrRoll);
  prevErrorRoll = errorRoll;

  // 8. Command the Servos 
  // IMPORTANT: If a servo moves the WRONG way, change the "+" to a "-" below!
  int servoPitchPos = 90 + outputPitch; 
  int servoRollPos = 90 + outputRoll;

  // Safety Constraint: Stop the servos from destroying the plastic bracket
  servoPitchPos = constrain(servoPitchPos, 30, 150);
  servoRollPos = constrain(servoRollPos, 30, 150);

  servoPitch.write(servoPitchPos);
  servoRoll.write(servoRollPos);

  // 9. Serial Output for the Live Graph Demo
  Serial.print("Target_0_Deg:0,");
  Serial.print("SpoonPitch:");
  Serial.println(pitch);
}
